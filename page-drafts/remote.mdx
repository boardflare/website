import { RemoteContent } from 'nextra/data'
import { buildDynamicMDX, buildDynamicMeta } from 'nextra/remote'

export const getStaticProps = async () => {
  const res = await fetch(
    'https://raw.githubusercontent.com/shuding/nextra/e2033531a53eeb8226e6ffcaab6e7e1667213566/examples/swr-site/pages/about/a-page.en-US.mdx'
  )
  const content = await res.text()
  if (!content) {
    throw new Error('Error while fetching data from GitHub.')
  }
  return {
    props: {
      ...(await buildDynamicMDX(content, {
        defaultShowCopyCode: true
      })),
      ...(await buildDynamicMeta())
    }
  }
}

<RemoteContent />
